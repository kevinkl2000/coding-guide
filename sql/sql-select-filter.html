Lesson 2

SELECT and Filtering

Filtering turns raw table reads into meaningful data retrieval.

WHERE Clause

SELECT id, name
FROM users
WHERE active = 1;

Common Operators

SELECT * FROM orders WHERE total > 100;
SELECT * FROM users WHERE country = 'US';
SELECT * FROM products WHERE stock <= 10;

AND, OR, IN, BETWEEN

SELECT * FROM users WHERE active = 1 AND role = 'admin';
SELECT * FROM users WHERE country IN ('US', 'CA', 'UK');
SELECT * FROM orders WHERE total BETWEEN 50 AND 200;
