Common SQL Commands

Reusable commands for day-to-day SQL querying and schema work. Click copy on any snippet.

SELECT

Read all rows

SELECT *
FROM table_name;
WHERE

Filter rows

SELECT id, name
FROM table_name
WHERE status = 'active';
ORDER

Sort and limit

SELECT *
FROM table_name
ORDER BY created_at DESC
LIMIT 20;
GROUP

Group and count

SELECT category, COUNT(*) AS total
FROM table_name
GROUP BY category;
JOIN

Join two tables

SELECT a.id, a.name, b.value
FROM table_a a
INNER JOIN table_b b ON b.a_id = a.id;
INSERT

Insert one row

INSERT INTO table_name (col1, col2)
VALUES ('value1', 'value2');
UPDATE

Update rows

UPDATE table_name
SET col1 = 'new_value'
WHERE id = 1;
DELETE

Delete rows

DELETE FROM table_name
WHERE id = 1;
CREATE

Create table

CREATE TABLE table_name (
    id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL
);
INDEX

Create index

CREATE INDEX idx_table_name_col
ON table_name (col1);
